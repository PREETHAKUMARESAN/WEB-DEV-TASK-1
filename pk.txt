<script type="text/javascript">          /*javascript begins*/
var rocketLaunch=new Date(" may 28, 2016 23:59:59");
var now=new Date();
var timestop = new Date(Date.parse(new Date()) + rocketLaunch.getTime() - now.getTime());

function remainingTime(end) 
{
  var t = Date.parse(end) - Date.parse(new Date());
  var seconds = Math.floor((t / 1000) % 60);
  var minutes = Math.floor((t / 1000 / 60) % 60);             /*to round off so that we dont have fractional seconds or minutes*/
  var hours = Math.floor((t / (1000 * 60 * 60)) % 24);
  var days = Math.floor(t / (1000 * 60 * 60 * 24));
  return {
    'total': t, 'days': days, 'hours': hours,'minutes': minutes,'seconds': seconds
  };
}

function initiateTime(id,end) 
{
  var timer= document.getElementById(id);
  var daysFinal = timer.querySelector('.days');           /*returns the first element that matches in the document*/
  var hoursFinal = timer.querySelector('.hours');          /*get the first element in clock where class=days*/
  var minutesFinal = timer.querySelector('.minutes');
  var secondsFinal = timer.querySelector('.seconds');

  function changeTime()
  {
    var t = remainingTime(end);
    daysFinal.innerHTML = t.days;
    hoursFinal.innerHTML = t.hours;
    minutesFinal.innerHTML = t.minutes;
    secondsFinal.innerHTML = t.seconds;

    if (t.total <= 0) 
	{
      clearInterval(time);
    }
  }

  changeTime();
  var time = setInterval(changeTime,1000);
}
initiateTime('demo',timestop);
</script>
